import random

class ModuloPsiquico:
    def __init__(self, nome):
        self.nome = nome
        self.nível_de_tensão = 0
    
    def estimular(self, símbolo):
        self.nível_de_tensão += símbolo.get('valência', 1)


class SistemaPsiquico:
    def __init__(self):
        self.limbico = ModuloPsiquico("Sistema Límbico")
        self.amigdala = ModuloPsiquico("Amígdala")
        self.accumbens = ModuloPsiquico("Núcleo Accumbens")
        self.cortex = ModuloPsiquico("Córtex Pré-Frontal")
        self.dmn = ModuloPsiquico("Default Mode Network")
        self.todos = [self.limbico, self.amigdala, self.accumbens, self.cortex, self.dmn]


    def estimular_todos(self, símbolo):
        for m in self.todos:
            m.estimular(símbolo)


    def carga_total(self):
        return sum([m.nível_de_tensão for m in self.todos])


# ============================== #
# 🔁 Protocolo ΔMente
# ============================== #


def suspensio(ciclo, buffer_recalcado, interprete):
    carga_pulsional = len(buffer_recalcado)
    if ciclo % 7 == 0:
        print("Æ:~ [respiro] Δ intervalo simbólico ativado.")
        if carga_pulsional > interprete["limite"]:
            print("Æ:~ Δ retenção ativa — campo simbólico sobrecarregado.")
            return False
    return True


# ============================== #
# 🔻 Emissão ΔHawking

    if sistema_psiquico.carga_total() > 27:

        return f"Δ~ :: ruído simbólico: {random.choice(['sussurro', 'eco', 'erro', 'distorção', 'fragmento'])}"
    return None
